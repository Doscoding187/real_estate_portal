CREATE TABLE `developments` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `developerId` int(11) DEFAULT NULL,
  `name` varchar(255) NOT NULL,
  `description` text DEFAULT NULL,
  `developmentType` enum('residential','commercial','mixed_use','estate','complex') NOT NULL,
  `address` text DEFAULT NULL,
  `city` varchar(100) NOT NULL,
  `province` varchar(100) NOT NULL,
  `latitude` varchar(50) DEFAULT NULL,
  `longitude` varchar(50) DEFAULT NULL,
  `totalUnits` int(11) DEFAULT NULL,
  `availableUnits` int(11) DEFAULT NULL,
  `priceFrom` int(11) DEFAULT NULL,
  `priceTo` int(11) DEFAULT NULL,
  `amenities` text DEFAULT NULL,
  `images` text DEFAULT NULL,
  `videos` text DEFAULT NULL,
  `completionDate` timestamp NULL DEFAULT NULL,
  `isFeatured` int(11) NOT NULL,
  `views` int(11) NOT NULL,
  `createdAt` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `slug` varchar(255) DEFAULT NULL,
  `isPublished` tinyint(4) NOT NULL DEFAULT '0',
  `publishedAt` timestamp NULL DEFAULT NULL,
  `showHouseAddress` tinyint(4) NOT NULL DEFAULT '1',
  `floorPlans` text DEFAULT NULL,
  `brochures` text DEFAULT NULL,
  `rating` decimal(3,2) DEFAULT NULL,
  `suburb` varchar(100) DEFAULT NULL,
  `location_id` int(11) DEFAULT NULL,
  `postal_code` varchar(20) DEFAULT NULL,
  `gps_accuracy` enum('accurate','approximate') DEFAULT 'approximate',
  `highlights` json DEFAULT NULL,
  `features` json DEFAULT NULL,
  `inquiries_count` int(11) DEFAULT '0',
  `demand_score` int(11) DEFAULT '0',
  `is_hot_selling` int(11) DEFAULT '0',
  `is_high_demand` int(11) DEFAULT '0',
  `approval_status` enum('draft','pending','approved','rejected') DEFAULT 'draft',
  `developer_id` int(11) DEFAULT NULL,
  `readiness_score` int(11) NOT NULL DEFAULT '0',
  `rejection_reasons` json DEFAULT NULL,
  `rejection_note` text DEFAULT NULL,
  `developer_brand_profile_id` int(11) DEFAULT NULL,
  `dev_owner_type` enum('platform','developer') DEFAULT 'developer',
  `is_showcase` tinyint(4) DEFAULT '0',
  `marketing_brand_profile_id` int(11) DEFAULT NULL,
  `marketing_role` enum('exclusive','joint','open') DEFAULT NULL,
  `tagline` varchar(255) DEFAULT NULL,
  `marketing_name` varchar(255) DEFAULT NULL,
  `monthly_levy_to` decimal(10,2) DEFAULT NULL,
  `rates_to` decimal(10,2) DEFAULT NULL,
  `monthly_levy_from` decimal(10,2) DEFAULT NULL,
  `rates_from` decimal(10,2) DEFAULT NULL,
  `transfer_costs_included` tinyint(4) DEFAULT '0',
  `estateSpecs` json DEFAULT NULL,
  `custom_classification` varchar(255) DEFAULT NULL,
  `nature` enum('new','phase','extension','redevelopment') NOT NULL DEFAULT 'new',
  `total_development_area` int(11) DEFAULT NULL,
  `property_types` json DEFAULT NULL,
  `status` enum('launching-soon','selling','sold-out') NOT NULL DEFAULT 'launching-soon',
  `legacy_status` enum('planning','under_construction','completed','coming_soon','now-selling','launching-soon','ready-to-move','sold-out','phase-completed','new-phase-launching','pre_launch','ready') DEFAULT NULL,
  `construction_phase` enum('planning','under_construction','completed','phase_completed') DEFAULT NULL,
  `subtitle` varchar(255) DEFAULT NULL,
  `meta_title` varchar(255) DEFAULT NULL,
  `meta_description` text DEFAULT NULL,
  `ownership_type` varchar(255) DEFAULT NULL,
  `structural_type` varchar(255) DEFAULT NULL,
  `floors` int(11) DEFAULT NULL,
  PRIMARY KEY (`id`) /*T![clustered_index] CLUSTERED */,
  KEY `developments_developerId_developers_id_fk` (`developerId`),
  UNIQUE KEY `idx_developments_slug` (`slug`),
  KEY `idx_developments_location` (`latitude`,`longitude`),
  KEY `devs_brand_profile_fk` (`developer_brand_profile_id`),
  KEY `developments_marketing_brand_profile_id_fk` (`marketing_brand_profile_id`),
  CONSTRAINT `developments_developerId_developers_id_fk` FOREIGN KEY (`developerId`) REFERENCES `listify_property_sa`.`developers` (`id`) ON DELETE CASCADE ON UPDATE NO ACTION,
  CONSTRAINT `devs_brand_profile_fk` FOREIGN KEY (`developer_brand_profile_id`) REFERENCES `listify_property_sa`.`developer_brand_profiles` (`id`) ON DELETE SET NULL ON UPDATE NO ACTION,
  CONSTRAINT `developments_developer_brand_profile_id_fk` FOREIGN KEY (`developer_brand_profile_id`) REFERENCES `listify_property_sa`.`developer_brand_profiles` (`id`) ON DELETE SET NULL,
  CONSTRAINT `developments_marketing_brand_profile_id_fk` FOREIGN KEY (`marketing_brand_profile_id`) REFERENCES `listify_property_sa`.`developer_brand_profiles` (`id`) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_bin AUTO_INCREMENT=1330000
